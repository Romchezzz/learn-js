## Задача/Вопрос 5
```js
function coinFlip(){
    let flip = Math.random();
      return flip <= 0.3 ? 'tails' : 'heads';
};

let heads = 0;
let tails = 0;

for (let i = 0; i < 1000; i++){
    if(coinFlip() === 'heads') heads++;
      else tails++;
};

let headsOdds = heads/10;
let tailsOdds = tails/10;

console.log (`Heads - ${headsOdds}% Tails - ${tailsOdds}%`);
```
